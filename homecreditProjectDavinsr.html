# Fungsi evaluasi outlier
def evaluate_outliers(data, dataset_name):
    print(f"=== Evaluasi Outlier untuk {dataset_name} ===")
    
    # Descriptive statistics
    print("\nDescriptive Statistics:")
    print(data.describe())

    # Boxplot untuk outlier
    plt.figure(figsize=(12, 6))
    for i, feature in enumerate(data.columns, 1):
        plt.subplot(1, len(data.columns), i)
        sns.boxplot(y=data[feature])
        plt.title(f"Boxplot - {feature}")
    plt.tight_layout()
    plt.show()

    # Identifikasi outlier menggunakan IQR
    Q1 = data.quantile(0.25)
    Q3 = data.quantile(0.75)
    IQR = Q3 - Q1

    outlier_condition = (data < (Q1 - 1.5 * IQR)) | (data > (Q3 + 1.5 * IQR))
    outlier_count = outlier_condition.sum().sum()
    outlier_percentage = (outlier_count / data.size) * 100

    print(f"\nJumlah outlier: {outlier_count}")
    print(f"Persentase outlier: {outlier_percentage:.2f}%")

    return outlier_condition

# Evaluasi untuk installments

# 3. Outlier Detection & Handling
def handle_outliers(df, columns):
    outlier_condition = pd.DataFrame()
    for column in columns:
        Q1 = df[column].quantile(0.25)
        Q3 = df[column].quantile(0.75)
        IQR = Q3 - Q1
        outlier_condition[column] = (df[column] < (Q1 - 1.5 * IQR)) | (df[column] > (Q3 + 1.5 * IQR))
    return ~outlier_condition.any(axis=1)

# 3. Buat fungsi untuk evaluasi model dengan cross validation
def evaluate_model(model, X, y, model_name):
    # Menggunakan KFold untuk cross validation
    kf = KFold(n_splits=5, shuffle=True, random_state=42)
    
    # Evaluasi dengan cross validation
    cv_scores = cross_val_score(model, X, y, cv=kf)
    
    print(f"\n{model_name} Cross Validation Scores:")
    print(f"Individual scores: {cv_scores}")
    print(f"Mean CV Score: {cv_scores.mean():.3f} (+/- {cv_scores.std() * 2:.3f})")

# 4. Coba beberapa model berbeda
# Pipeline untuk LogisticRegression
lr_pipeline = Pipeline([
    ('scaler', StandardScaler()),
    ('classifier', LogisticRegression(random_state=42))
])

# Pipeline untuk SVC
svc_pipeline = Pipeline([
    ('scaler', StandardScaler()),
    ('classifier', SVC(kernel='rbf', random_state=42))
])

# Pipeline untuk RandomForest (tidak perlu scaling)
rf_pipeline = Pipeline([
    ('classifier', RandomForestClassifier(n_estimators=50, max_depth=5, random_state=42))
])

# 5. Evaluasi semua model
models = {
    'Logistic Regression': lr_pipeline,
    'Support Vector Machine': svc_pipeline,
    'Random Forest': rf_pipeline
}

for name, model in models.items():
    evaluate_model(model, X, y, name)
